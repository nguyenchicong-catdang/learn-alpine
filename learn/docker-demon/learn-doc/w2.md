# Mặc định khi gõ 'make' sẽ hiện hướng dẫn
.DEFAULT_GOAL := help

help:
	@echo "Các lệnh hỗ trợ:"
	@echo "  make build   - Build image Alpine"
	@echo "  make up      - Chạy docker ngầm (lách luật kernel)"
	@echo "  make clean   - Dọn dẹp hệ thống"

build:
	docker build -t my-app .

up:
	# Dùng --net=host để né lỗi NAT/Bridge của Kernel
	docker run -d --name my-app --net=host my-app

clean:
	-docker stop my-app
	-docker rm my-app
