define DOCKERFILE_CONTENT
FROM alpine:3.18
RUN apk add --no-cache nginx
COPY . /usr/share/nginx/html
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
endef

nginx-build:
	mkdir -p $(PATH_NGINX)
	@echo "$$DOCKERFILE_CONTENT" > $(PATH_NGINX)/Dockerfile
	@echo "Đã tạo Dockerfile thành công bằng Heredoc!"



nginx-build:
	mkdir -p $(PATH_NGINX)
	@cat <<EOF > $(PATH_NGINX)/Dockerfile
FROM alpine:latest
RUN apk add --no-cache curl
# Đây là comment bên trong Dockerfile
ENV APP_ENV=$(APP_ENV)
EOF
	@echo "Xong!"
