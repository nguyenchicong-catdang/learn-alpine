# ... (đoạn RUN apk add giữ nguyên)

# Tạo thư mục và chép file vào
RUN mkdir -p /var/www/localhost/htdocs /run/nginx

# Chép config và index từ máy host (do Makefile tạo ra) vào image
COPY default.conf /etc/nginx/http.d/default.conf
COPY index.html /var/www/localhost/htdocs/index.html

# Đảm bảo quyền sở hữu
RUN chown -R nginx:nginx /var/www/localhost/htdocs

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]


docker run -d --name $(IMAGE_NAME) \
    --network host \
    -v $(PATH_NGINX)/index.html:/var/www/localhost/htdocs/index.html \
    -v $(PATH_NGINX)/default.conf:/etc/nginx/http.d/default.conf \
    $(IMAGE_NAME)


FROM $(IMAGE_ALPINE)
RUN apk add --no-cache nginx && mkdir -p /run/nginx

# Đặt thư mục làm việc chính
WORKDIR /var/www/localhost/htdocs

# Giờ chỉ cần COPY vào dấu chấm (.)
COPY index.html .
COPY default.conf /etc/nginx/http.d/default.conf

# Lệnh chạy vẫn vậy
CMD ["nginx", "-g", "daemon off;"]
